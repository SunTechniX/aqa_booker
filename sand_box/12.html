🔍 Часть 1: Как правильно тестировать GraphQL (принципы для SDET)

✅ 1. GraphQL ≠ REST: ключевые отличия в тестировании

Аспект Особенности для тестировщика
Один эндпоинт Все запросы идут на POST /graphql. Тестируем не URL, а тело запроса (query/mutation).
HTTP 200 при ошибках Сервер почти всегда возвращает 200 OK. Ошибки — в поле errors ответа. Проверяем и data, и errors.
Строгая схема Контракт описан в SDL. Можно автоматически валидировать запросы, генерировать тесты, проверять совместимость.
Частичные ответы Если одно поле упало с ошибкой, остальные могут вернуться в data. Тестируем сценарии partial success.
Переменные и фрагменты Не хардкодьте значения в query. Используйте variables — это чище, безопаснее и удобнее для параметризации.

✅ 3. Чек-лист тестирования (мини-гайд)

  • Запрос валидируется сервером (синтаксис, типы аргументов)
  • HTTP 200 + проверка: есть ли errors? соответствует ли data ожидаемой структуре?
  • Запрос несуществующего order(id: 999) → data.order == null, без errors
  • create_order возвращает новый id и статус created
  • change_status разрешает только переходы из ALLOWED_TRANSITIONS
  • Недопустимый переход → errors с понятным сообщением, data.change_status == null
  • Повторный create_order с теми же данными → создаётся новый заказ (не дубликат)
  • Параллельные запросы не ломают консистентность (опционально, нагрузочный тест)
  • Схема не «утекла» в продакшен через интроспекцию (если так задумано)

tests/utils.py — хелпер для отправки запросов 🔧

import requests
from typing import Optional, Dict, Any

GRAPHQL_URL = "http://localhost:8001/graphql"

def run_graphql(
    query: str,
    variables: Optional[Dict[str, Any]] = None,
    operation_name: Optional[str] = None,
    url: str = GRAPHQL_URL
) -> Dict[str, Any]:
    """Универсальная функция для выполнения GraphQL-запроса."""
    payload = {"query": query.strip()}
    if variables:
        payload["variables"] = variables
    if operation_name:
        payload["operationName"] = operation_name
    
    response = requests.post(url, json=payload, headers={"Content-Type": "application/json"})
    response.raise_for_status()  # Выбросит исключение при 4xx/5xx (редко для GraphQL)
    return response.json()

def assert_graphql_success(response: Dict[str, Any], path: Optional[str] = None):
    """Проверка, что в ответе нет ошибок и данные на месте."""
    assert "errors" not in response, f"Unexpected GraphQL errors: {response.get('errors')}"
    assert "data" in response, "Response missing 'data' field"
    
    if path:
        # Поддержка вложенных путей: "order" или "user.address.city"
        value = response["data"]
        for key in path.split("."):
            assert value is not None, f"Path '{path}' broken at key '{key}': value is None"
            value = value.get(key)
        assert value is not None, f"Value at path '{path}' is None"

tests/test_queries.py — тесты на чтение данных 📥

import pytest
from utils import run_graphql, assert_graphql_success

class TestOrdersQuery:
    def test_orders_returns_list(self):
        query = """
        query {
            orders {
                id
                customer
                status
            }
        }
        """
        response = run_graphql(query)
        assert_graphql_success(response)
        assert isinstance(response["data"]["orders"], list)

    def test_orders_empty_when_no_data(self, cleanup_orders):
        query = "{ orders { id customer status } }"
        response = run_graphql(query)
        assert response["data"]["orders"] == []

    def test_order_by_id_found(self, sample_order):
        query = """
        query GetOrder($id: Int!) {
            order(id: $id) {
                id
                customer
                status
            }
        }
        """
        response = run_graphql(query, variables={"id": sample_order.id})
        assert_graphql_success(response, path="order")
        data = response["data"]["order"]
        assert data["id"] == sample_order.id
        assert data["customer"] == sample_order.customer
        assert data["status"] == sample_order.status

    def test_order_by_id_not_found(self):
        query = "query($id: Int!) { order(id: $id) { id } }"
        response = run_graphql(query, variables={"id": 99999})
        assert_graphql_success(response)  # Нет errors, но data.order == null
        assert response["data"]["order"] is None

tests/test_mutations.py — тесты на изменение данных ✏️

import pytest
from utils import run_graphql, assert_graphql_success
from schema import STATUS_CREATED, STATUS_PAID, STATUS_CANCELLED, STATUS_SHIPPED, STATUS_DELIVERED

class TestCreateOrder:
    def test_create_order_success(self):
        mutation = """
        mutation CreateOrder($customer: String!) {
            createOrder(customer: $customer) {
                id
                customer
                status
            }
        }
        """
        response = run_graphql(mutation, variables={"customer": "Alice"})
        assert_graphql_success(response, path="createOrder")
        order = response["data"]["createOrder"]
        assert order["customer"] == "Alice"
        assert order["status"] == STATUS_CREATED
        assert isinstance(order["id"], int) and order["id"] > 0

    def test_create_order_required_field_missing(self):
        mutation = "mutation { createOrder { id } }"  # нет аргумента customer
        response = run_graphql(mutation)
        assert "errors" in response
        assert any("customer" in err["message"].lower() for err in response["errors"])

class TestChangeStatus:
    @pytest.mark.parametrize("from_status, to_status, should_succeed", [
        (STATUS_CREATED, STATUS_PAID, True),
        (STATUS_CREATED, STATUS_CANCELLED, True),
        (STATUS_PAID, STATUS_SHIPPED, True),
        (STATUS_PAID, STATUS_CANCELLED, True),
        (STATUS_SHIPPED, STATUS_DELIVERED, True),
        (STATUS_CREATED, STATUS_SHIPPED, False),  # пропуск этапа
        (STATUS_PAID, STATUS_DELIVERED, False),    # нельзя перепрыгнуть
        (STATUS_DELIVERED, STATUS_CANCELLED, False), # финальный статус
        (STATUS_CANCELLED, STATUS_PAID, False),    # отменённый нельзя изменить
    ])
    def test_status_transitions(self, db_session, from_status, to_status, should_succeed):
        # Создаём заказ с начальным статусом
        from schema import Order
        order = Order(customer="Transition Tester", status=from_status)
        db_session.add(order)
        db_session.commit()
        db_session.refresh(order)
        
        mutation = """
        mutation ChangeStatus($id: Int!, $status: String!) {
            changeStatus(id: $id, status: $status) {
                id
                status
            }
        }
        """
        response = run_graphql(mutation, variables={"id": order.id, "status": to_status})
        
        if should_succeed:
            assert_graphql_success(response, path="changeStatus")
            assert response["data"]["changeStatus"]["status"] == to_status
            # Проверяем, что в БД тоже обновилось
            db_session.refresh(order)
            assert order.status == to_status
        else:
            assert "errors" in response
            assert any("cannot change status" in err["message"].lower() for err in response["errors"])
            # Статус не должен измениться
            db_session.refresh(order)
            assert order.status == from_status

    def test_change_status_order_not_found(self):
        mutation = "mutation($id: Int!, $status: String!) { changeStatus(id: $id, status: $status) { id } }"
        response = run_graphql(mutation, variables={"id": 99999, "status": "paid"})
        assert "errors" in response
        assert any("not found" in err["message"].lower() for err in response["errors"])

🔹 3. Нагрузочное тестирование (k6 + GraphQL)

// load-test/graphql.js
import http from 'k6/http';
import { check } from 'k6';

export const options = { vus: 10, duration: '30s' };

export default function () {
  const query = `query { orders { id customer status } }`;
  const res = http.post('http://localhost:8001/graphql', 
    JSON.stringify({ query }),
    { headers: { 'Content-Type': 'application/json' }}
  );
  
  check(res, {
    'status is 200': (r) => r.status === 200,
    'no graphql errors': (r) => !r.json('errors'),
    'response time < 200ms': (r) => r.timings.duration < 200,
  });
}

Запуск: k6 run load-test/graphql.js

💡 Главный совет: В GraphQL вы тестируете контракт и поведение, а не эндпоинты.
Фокус на: схема → переменные → ошибки в errors → бизнес-правила → частичные ответы.
