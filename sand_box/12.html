Примеры для обучения (pytest + requests)

🔐 Получение сессионного токена (полный флоу)

# test_auth_flow.py
import requests

BASE = "https://api.reqres.in"
PUBLIC_KEY = "pk_test_..."  # из настроек проекта
PROJECT_ID = "proj_..."

def test_login_verify_flow():
    # 1. Запрашиваем код входа
    resp = requests.post(
        f"{BASE}/api/app-users/login",
        headers={"x-api-key": PUBLIC_KEY, "Content-Type": "application/json"},
        json={"email": "student@example.com", "project_id": PROJECT_ID}
    )
    assert resp.status_code == 200
    assert resp.json()["data"]["sent"] is True
    
    # 2. В тестовом режиме токен возвращается сразу (MAGIC_LINK_TEST_MODE)
    # В продакшене — код придёт на email
    token = resp.json().get("data", {}).get("token")  # только в тестовом режиме!
    
    # 3. Верифицируем код → получаем session_token
    resp = requests.post(
        f"{BASE}/api/app-users/verify",
        headers={"Content-Type": "application/json"},
        json={"token": token}
    )
    assert resp.status_code == 200
    session_token = resp.json()["session_token"]
    
    # 4. Используем session_token для авторизованных запросов
    resp = requests.get(
        f"{BASE}/api/app-users/me",
        headers={"Authorization": f"Bearer {session_token}"}
    )
    assert resp.status_code == 200
    assert resp.json()["email"] == "student@example.com"

📦 Работа с коллекциями (CRUD)

# test_collections.py
import pytest
import requests

BASE = "https://api.reqres.in"
API_KEY = "sk_test_..."  # Project API Key (admin)
PROJECT_ID = "proj_..."

@pytest.fixture
def admin_headers():
    return {"x-api-key": API_KEY, "Content-Type": "application/json"}

def test_collection_crud(admin_headers):
    # CREATE коллекцию
    resp = requests.post(
        f"{BASE}/api/collections",
        headers=admin_headers,
        json={
            "name": "Student Tasks",
            "slug": "tasks",
            "project_id": PROJECT_ID,
            "visibility": "private",
            "schema": {
                "type": "object",
                "properties": {
                    "title": {"type": "string"},
                    "done": {"type": "boolean"}
                }
            }
        }
    )
    assert resp.status_code == 201
    collection_id = resp.json()["id"]
    
    # CREATE запись
    resp = requests.post(
        f"{BASE}/api/collections/tasks/records",
        headers=admin_headers,
        params={"project_id": PROJECT_ID},
        json={"data": {"title": "Learn mocking", "done": False}}
    )
    assert resp.status_code == 201
    record_id = resp.json()["id"]
    
    # READ записи
    resp = requests.get(
        f"{BASE}/api/collections/tasks/records",
        headers=admin_headers,
        params={"project_id": PROJECT_ID, "limit": 10}
    )
    assert resp.status_code == 200
    data = resp.json()["data"]
    assert any(r["id"] == record_id for r in data)
    
    # UPDATE запись (через админ-эндпоинт)
    resp = requests.put(  # или PATCH, если поддерживается
        f"{BASE}/api/collections/tasks/records/{record_id}",
        headers=admin_headers,
        params={"project_id": PROJECT_ID},
        json={"data": {"title": "Learn mocking", "done": True}}
    )
    assert resp.status_code in (200, 204)
    
    # DELETE запись (если эндпоинт доступен)
    # resp = requests.delete(...)
    # assert resp.status_code in (200, 204)

⚠️ Ограничения и нюансы (май 2026)

  1. Все запросы требуют x-api-key — даже публичные. Получите ключи в app.reqres.in.
  2. Два типа ключей:
    • Project API Key (секретный) — для бэкенда, админ-операций.
    • Public Project Key — для фронтенда, безопасен для клиента.
  3. Сессионные токены живут 15 минут по умолчанию — обрабатывайте 401 и делайте ре-логин.
  4. Профили пользователей возвращаются только в админ-эндпоинтах (GET /api/app-users/{id}), но не в /me.
  5. Коллекции могут быть public или private — публичные доступны для чтения без авторизации (с x-api-key: public_key).
  6. Rate limits: 1000 запросов/час на бесплатном тарифе — отслеживайте заголовки x-ratelimit-*.
